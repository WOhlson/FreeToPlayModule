#Creating a new table
CREATE TABLE Organizer
(
  O_ID INT NOT NULL,
  Organizer_name VARCHAR(50) NOT NULL,
  PRIMARY KEY (O_ID)
);

CREATE TABLE Tournament
(
  T_ID INT NOT NULL,
  Tment_Name VARCHAR(50) NOT NULL,
  O_ID INT NOT NULL,
  PRIMARY KEY (T_ID),
  FOREIGN KEY (O_ID) REFERENCES Organizer(O_ID)
);

CREATE TABLE Matches
(
  Round INT NOT NULL,
  Winner VARCHAR(20) NOT NULL,
  Date VARCHAR(20) NOT NULL,
  Hour INT NOT NULL,
  Minute INT NOT NULL,
  M_ID INT NOT NULL,
  T_ID INT NOT NULL,
  PRIMARY KEY (M_ID),
  FOREIGN KEY (T_ID) REFERENCES Tournament(T_ID)
);

CREATE TABLE Tournament_Dates
(
  Dates VARCHAR(20) NOT NULL,
  T_ID INT NOT NULL,
  PRIMARY KEY (Dates, T_ID),
  FOREIGN KEY (T_ID) REFERENCES Tournament(T_ID)
);

CREATE TABLE Player
(
  Password VARCHAR(20) NOT NULL,
  Username VARCHAR(20) NOT NULL,
  Display_Name VARCHAR(20) NOT NULL,
  Seed INT NOT NULL,
  P_ID INT NOT NULL,
  T_ID INT NOT NULL,
  M_ID INT NOT NULL,
  PRIMARY KEY (P_ID),
  FOREIGN KEY (T_ID) REFERENCES Tournament(T_ID),
  FOREIGN KEY (M_ID) REFERENCES Matches(M_ID)
);
